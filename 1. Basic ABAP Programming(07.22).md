- `AT END OF` : 같은 값으로 묶인 데이터가 끝나는 시점(마지막 행)에서 실행되는 Control Break 문법
```abap
AT END OF <field>.
  ...
ENDAT.
```

CLASS zcl_a06_w04 IMPLEMENTATION.
  METHOD if_oo_adt_classrun~main.
    TYPES: BEGIN OF ts_data,
             month(2),
             mcount TYPE I,
           END OF ts_data,
           tt_data type table of ts_data.
    DATA: lt_data TYPE tt_data,
          lt_sum LIKE lt_data.
    lt_data = value #(
                        ( month = '01' mcount = '10') ( month = '02' mcount = '20') ( month = '03' mcount = '30') ( month = '04' mcount = '40')
                        ( month = '01' mcount = '20') ( month = '02' mcount = '40') ( month = '03' mcount = '30') ( month = '04' mcount = '40')
                    ).
    SORT lt_data BY month.

    LOOP AT lt_data INTO DATA(ls_data).
      AT END OF month.
        SUM.
        APPEND ls_data TO lt_sum.
      ENDAT.
    ENDLOOP.
    out->write(  lt_sum ).
  ENDMETHOD.
ENDCLASS.

CLASS zcl_a06ex01 IMPLEMENTATION.
  METHOD set_book.
    SELECT FROM sbook
        FIELDS substring( fldate, 1, 4 ) AS year, substring( fldate, 5, 2 ) AS month, carrid, class, cancelled
        WHERE carrid = @pa_carr
        INTO CORRESPONDING FIELDS OF TABLE @lt_temp.

    SELECT FROM sbook
        FIELDS DISTINCT substring( fldate, 1, 4 ) AS year, carrid, class
        WHERE carrid = @pa_carr
        INTO CORRESPONDING FIELDS OF TABLE @lt_book.

    DATA: lv_res TYPE i,
          lv_can TYPE i.
    SORT lt_temp BY year ASCENDING month ASCENDING class.
    SORT lt_book BY year ASCENDING class.

    LOOP AT lt_temp ASSIGNING FIELD-SYMBOL(<ls_temp>).
      IF <ls_temp>-cancelled = 'X'.
        <ls_temp>-can = 1.
        <ls_temp>-res = 0.
      ELSE.
        <ls_temp>-res = 1.
        <ls_temp>-can = 0.
      ENDIF.
    ENDLOOP.

    FIELD-SYMBOLS: <ls_book> TYPE ts_sbook.

    LOOP AT lt_temp INTO ls_temp.
      AT END OF month.
        SUM.
        READ TABLE lt_book ASSIGNING <ls_book>
          WITH KEY year   = ls_temp-year
                   carrid = ls_temp-carrid
                   class  = ls_temp-class.

        IF sy-subrc = 0.
          CASE ls_temp-month.
            WHEN '01'.
              <ls_book>-res_1 = ls_temp-res.
              <ls_book>-can_1 = ls_temp-can.
            WHEN '02'.
              <ls_book>-res_2 = ls_temp-res.
              <ls_book>-can_2 = ls_temp-can.
            WHEN '03'.
              <ls_book>-res_3 = ls_temp-res.
              <ls_book>-can_3 = ls_temp-can.
            WHEN '04'.
              <ls_book>-res_4 = ls_temp-res.
              <ls_book>-can_4 = ls_temp-can.
            WHEN '05'.
              <ls_book>-res_5 = ls_temp-res.
              <ls_book>-can_5 = ls_temp-can.
            WHEN '06'.
              <ls_book>-res_6 = ls_temp-res.
              <ls_book>-can_6 = ls_temp-can.
            WHEN '07'.
              <ls_book>-res_7 = ls_temp-res.
              <ls_book>-can_7 = ls_temp-can.
            WHEN '08'.
              <ls_book>-res_8 = ls_temp-res.
              <ls_book>-can_8 = ls_temp-can.
            WHEN '09'.
              <ls_book>-res_9 = ls_temp-res.
              <ls_book>-can_9 = ls_temp-can.
            WHEN '10'.
              <ls_book>-res_10 = ls_temp-res.
              <ls_book>-can_10 = ls_temp-can.
            WHEN '11'.
              <ls_book>-res_11 = ls_temp-res.
              <ls_book>-can_11 = ls_temp-can.
            WHEN '12'.
              <ls_book>-res_12 = ls_temp-res.
              <ls_book>-can_12 = ls_temp-can.
          ENDCASE.
        ENDIF.
        UNASSIGN <ls_book>.
      ENDAT.
    ENDLOOP.
  ENDMETHOD.
ENDCLASS.
