```abap
CLASS zcl_a06ex02 DEFINITION
  PUBLIC
  FINAL
  CREATE PUBLIC .

  PUBLIC SECTION.
    TYPES: BEGIN OF ts_temp,
             year      TYPE c LENGTH 4,
             carrid    TYPE sbook-carrid,
             class     TYPE sbook-class,
             month(2),
             cancelled TYPE sbook-cancelled,
             res       TYPE i,
             can       TYPE i,
           END OF ts_temp.
    TYPES tt_temp TYPE TABLE OF ts_temp.
    DATA: lt_temp  TYPE tt_temp,
          lt_temp2 LIKE lt_temp.

    TYPES: BEGIN OF ts_sbook,
             year      TYPE c LENGTH 4,
             carrid    TYPE sbook-carrid,
             carrname  TYPE scarr-carrname,
             classname TYPE zcds_a06_ex12-Text,
             class     TYPE sbook-class,
             res01     TYPE i, can01   TYPE i,
             res02     TYPE i, can02   TYPE i,
             res03     TYPE i, can03   TYPE i,
             res04     TYPE i, can04   TYPE i,
             res05     TYPE i, can05   TYPE i,
             res06     TYPE i, can06   TYPE i,
             res07     TYPE i, can07   TYPE i,
             res08     TYPE i, can08   TYPE i,
             res09     TYPE i, can09   TYPE i,
             res10     TYPE i, can10   TYPE i,
             res11     TYPE i, can11   TYPE i,
             res12     TYPE i, can12   TYPE i,
           END OF ts_sbook.
    TYPES tt_sbook TYPE TABLE OF ts_sbook.

    METHODS set_book
      IMPORTING pa_carr TYPE sbook-carrid
      EXPORTING lt_book TYPE tt_sbook.
  PROTECTED SECTION.
  PRIVATE SECTION.
ENDCLASS.

CLASS zcl_a06ex02 IMPLEMENTATION.
  METHOD set_book.
    SELECT FROM sbook
        FIELDS substring( fldate, 1, 4 ) AS year, carrid, class, substring( fldate, 5, 2 ) AS month, cancelled
        WHERE carrid = @pa_carr
        INTO CORRESPONDING FIELDS OF TABLE @lt_temp.

    SELECT FROM sbook
        FIELDS DISTINCT substring( fldate, 1, 4 ) AS year, carrid, class
        WHERE carrid = @pa_carr
        INTO CORRESPONDING FIELDS OF TABLE @lt_book.

    SORT lt_temp BY year class month.
    SORT lt_book BY year class.

    DATA: lv_res TYPE i,
          lv_can TYPE i.

    LOOP AT lt_temp ASSIGNING FIELD-SYMBOL(<ls_temp>).
      IF <ls_temp>-cancelled = 'X'.
        <ls_temp>-can = 1.
        <ls_temp>-res = 0.
      ELSE.
        <ls_temp>-res = 1.
        <ls_temp>-can = 0.
      ENDIF.
    ENDLOOP.

    LOOP AT lt_temp INTO DATA(ls_temp).
      AT END OF month.
        SUM.
        APPEND ls_temp TO lt_temp2.
      ENDAT.
    ENDLOOP.

    FIELD-SYMBOLS: <fs_book>  TYPE any,
                   <fs_month> TYPE i,
                   <fs_scarr> TYPE any,
                   <fs_class> TYPE any.

    SELECT FROM scarr
        FIELDS carrid, carrname
        WHERE carrid = @pa_carr
        INTO TABLE @DATA(lt_scarr).

    DATA lt_dd07v TYPE TABLE OF dd07v.

    CALL FUNCTION 'DD_DOMVALUES_GET'
      EXPORTING
        domname        = 'S_CLASS'
        text           = 'X'
        langu          = sy-langu
      TABLES
        dd07v_tab      = lt_dd07v
      EXCEPTIONS
        wrong_textflag = 1
        OTHERS         = 2.

    LOOP AT lt_temp2 INTO DATA(ls_temp2).
      READ TABLE lt_book ASSIGNING <fs_book>
        WITH KEY year   = ls_temp2-year
                 carrid = ls_temp2-carrid
                 class  = ls_temp2-class.
      IF sy-subrc = 0.
        READ TABLE lt_scarr INTO DATA(ls_scarr) WITH KEY carrid = ls_temp2-carrid.
        ASSIGN COMPONENT 'CARRNAME' OF STRUCTURE <fs_book> TO <fs_scarr>.
        <fs_scarr> = ls_scarr-carrname.

        READ TABLE lt_dd07v INTO DATA(ls_class) WITH KEY domvalue_l = ls_temp2-class.
        ASSIGN COMPONENT 'CLASSNAME' OF STRUCTURE <fs_book> TO <fs_class>.
        <fs_class> = ls_class-ddtext.

        DATA(lv_res_month) = |RES{ ls_temp2-month }|.
        ASSIGN COMPONENT lv_res_month OF STRUCTURE <fs_book> TO <fs_month>.
        <fs_month> = ls_temp2-res.

        DATA(lv_can_month) = |CAN{ ls_temp2-month }|.
        ASSIGN COMPONENT lv_can_month OF STRUCTURE <fs_book> TO <fs_month>.
        <fs_month> = ls_temp2-can.
      ENDIF.
    ENDLOOP.
  ENDMETHOD.
ENDCLASS.
```
