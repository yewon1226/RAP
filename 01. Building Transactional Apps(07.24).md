### 2) Root View, Projection View 생성
- Root View 생성
<img width="488" height="313" alt="image" src="https://github.com/user-attachments/assets/82812f4a-3fe5-47ac-b7e4-2cc8c3e3d6fb" />

<img width="492" height="230" alt="image" src="https://github.com/user-attachments/assets/9dc11572-8b02-4c4d-870f-9b8644ee12fc" />

```abap
@AccessControl.authorizationCheck: #NOT_REQUIRED
@EndUserText.label: '[A06] 사원 Root View'
@Metadata.ignorePropagatedAnnotations: true
define root view entity ZR_EMP_A06 as select from zta06emp
{
    key pernr as PersonNumber,
    ename as EmployeeName,
    gesch as Gender,
    created_by as CreatedBy,
    creation_at as CreationAt,
    last_changed_by as LastChangedBy,
    last_changed_at as LastChangedAt
}
```

```abap
managed implementation in class zbp_r_emp_a06 unique;
strict ( 2 );

define behavior for ZR_EMP_A06 //alias <alias_name>
persistent table zta06emp
lock master
authorization master ( instance )
//etag master <field_name>
{
  create;
  update;
  delete;

  mapping for zta06emp
    {
      PersonNumber  = pernr;
      EmployeeName  = ename;
      Gender        = gesch;
      CreationAt    = creation_at;
      CreatedBy     = created_by;
      LastChangedAt = last_changed_at;
      LastChangedBy = last_changed_by;
    }
}
```

### 6) Projection View 수정 (디테일 창)
- Projection View 수정
- `provider contract transactional_query` 안써도 돌아감. 하지만 나중에 오류날 수 있으니 쓰는게 좋음.

```abap
@AccessControl.authorizationCheck: #NOT_REQUIRED
@EndUserText.label: '[A06] 사원 Projection View 2'
@Metadata.ignorePropagatedAnnotations: true
@Metadata.allowExtensions: true
define root view entity ZC_EMP_A06 
//provider contract transactional_query
as projection on ZR_EMP_A06
{
    key PersonNumber,
    EmployeeName,
    Gender,
    CreatedBy,
    CreationAt,
    LastChangedBy,
    LastChangedAt
}
```
