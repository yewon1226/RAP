```abap
@AccessControl.authorizationCheck: #NOT_REQUIRED
@EndUserText.label: '[A06] 사원 Projection View 2'
@Metadata.ignorePropagatedAnnotations: true
@Metadata.allowExtensions: true
define root view entity ZC_EMP_A06 

as projection on ZR_EMP_A06
{
    key PersonNumber,
    EmployeeName,
    Gender,
    CreatedBy,
    CreationAt,
    LastChangedBy,
    LastChangedAt
}
```
